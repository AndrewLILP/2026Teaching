Part 1 · Starter Scripts (copy to Unity)

Your Three Starter Scripts

Copy each script below into your Unity project's Assets/Scripts/ folder. Do not rename the files — the filename must exactly match the class name or Unity will refuse to compile.

Before you copy: In Unity, right-click the Scripts folder in the Project window → Create → C# Script. Name it exactly as shown (e.g. BoardSpace). Then open it, select all, delete all, and paste the code below. Save with Ctrl+S.
Script 1 of 3 — Board Space BoardSpace.cs 
using UnityEngine;

// Defines the four possible types of board space.
// Using an enum prevents typos (e.g. "Proprety") at compile time.
public enum SpaceType
{
    Go,
    Property,
    Tax,
    Chance
}

// Attach this script to every space GameObject on the board.
// Fill in the Inspector fields for each space — no code changes needed.
public class BoardSpace : MonoBehaviour
{
    // ── Identity ─────────────────────────────────────────────
    [Header("Space Identity")]
    public string    spaceName;      // e.g. "Harbour St"
    public SpaceType spaceType;      // set via dropdown in Inspector
    public Color     groupColour;    // property group colour

    // ── Property values (only used when spaceType == Property) ─
    [Header("Property Values")]
    public int       purchasePrice;  // cost to buy this property
    public int       rentAmount;     // rent charged to other players
    public Player    owner;          // null = unowned; set when purchased

    // ── Tax value (only used when spaceType == Tax) ────────────
    [Header("Tax")]
    public int       taxAmount;      // amount deducted when landed on

    // ── Visual reference ───────────────────────────────────────
    [Header("Visuals")]
    public Renderer  spaceRenderer;  // drag the MeshRenderer here

    // Called when the scene starts — tints the space with its group colour.
    void Start()
    {
        if (spaceRenderer != null && spaceType == SpaceType.Property)
        {
            // Create a new material instance so we don't change all spaces at once
            spaceRenderer.material = new Material(spaceRenderer.material);
            spaceRenderer.material.color = groupColour;
        }
    }

    // Returns true if this space can be purchased right now.
    public bool IsAvailable()
    {
        return spaceType == SpaceType.Property && owner == null;
    }

    // Returns true if another player owns this space (i.e. rent is owed).
    public bool IsOwnedByOther(Player currentPlayer)
    {
        return spaceType == SpaceType.Property
            && owner != null
            && owner != currentPlayer;
    }
}
Script 2 of 3 — Player Player.cs 
using UnityEngine;
using System.Collections.Generic;

// Attach this script to each player token GameObject.
// Set playerName, money, and token in the Inspector.
public class Player : MonoBehaviour
{
    // ── Identity ─────────────────────────────────────────────
    [Header("Player Info")]
    public string          playerName  = "Player 1";
    public Color           playerColour;           // token tint colour

    // ── Game state ───────────────────────────────────────────
    [Header("Game State")]
    public int             money       = 1500;     // starting funds
    public int             position    = 0;        // index into boardSpaces[]
    public bool            isActive    = true;     // false = eliminated

    // ── Owned properties ─────────────────────────────────────
    [Header("Properties")]
    public List<BoardSpace> ownedProperties = new List<BoardSpace>();

    // ── Scene references ─────────────────────────────────────
    [Header("Scene References")]
    public GameObject      token;      // the visible capsule/sphere in scene
    public Renderer        tokenRenderer;  // drag MeshRenderer here for tinting

    void Start()
    {
        // Tint the token to this player's colour
        if (tokenRenderer != null)
        {
            tokenRenderer.material = new Material(tokenRenderer.material);
            tokenRenderer.material.color = playerColour;
        }
    }

    // Adds money to this player (e.g. Go salary, selling property).
    public void AddMoney(int amount)
    {
        money += amount;
        Debug.Log($"{playerName} received ${amount}. Total: ${money}");
    }

    // Deducts money. Returns true if the player can afford it.
    public bool ChargeMoney(int amount)
    {
        money -= amount;
        Debug.Log($"{playerName} paid ${amount}. Remaining: ${money}");
        return money >= 0;  // false = bankrupt
    }

    // Adds a purchased property to this player's list.
    public void AddProperty(BoardSpace space)
    {
        space.owner = this;
        ownedProperties.Add(space);
        Debug.Log($"{playerName} now owns {space.spaceName}");
    }
}
Script 3 of 3 — Game Manager (setup stub) GameManager.cs 
using UnityEngine;
using System.Collections.Generic;
using TMPro;  // TextMeshPro — needed for UI text

// Central controller — attach to an empty GameObject called "GameManager".
// Wire up boardSpaces[], activePlayers, and UI references in the Inspector.
public class GameManager : MonoBehaviour
{
    // ── Board ────────────────────────────────────────────────
    [Header("Board")]
    public BoardSpace[] boardSpaces;    // drag ALL spaces in ORDER here

    // ── Players ──────────────────────────────────────────────
    [Header("Players")]
    public List<Player>  activePlayers;  // drag player GameObjects here
    private int          currentIndex = 0;

    // ── Rules ────────────────────────────────────────────────
    [Header("Rules")]
    public int goSalary = 200;

    // ── UI References ────────────────────────────────────────
    // We'll wire these up fully in Lesson 7.
    [Header("UI")]
    public TMP_Text messageText;       // drag MessagePanel/Text here
    public TMP_Text currentPlayerText; // drag HUD player name text here
    public TMP_Text currentMoneyText;  // drag HUD money text here

    // ── State ────────────────────────────────────────────────
    private bool gameOver = false;
    private bool waitingForInput = false; // blocks rolling mid-action

    void Start()
    {
        // Confirm the scene is wired correctly before anything runs
        if (boardSpaces == null || boardSpaces.Length == 0)
            Debug.LogError("GameManager: boardSpaces array is empty! "
                          + "Drag all board space GameObjects into the Inspector.");

        if (activePlayers == null || activePlayers.Count == 0)
            Debug.LogError("GameManager: No players assigned!");

        UpdateHUD();
        ShowMessage("Welcome to Dice District! "
                   + activePlayers[0].playerName
                   + " goes first. Roll the dice!");
    }

    // ── Dice ─────────────────────────────────────────────────
    public int RollDice()
    {
        int d1 = Random.Range(1, 7);
        int d2 = Random.Range(1, 7);
        Debug.Log($"Dice: {d1} + {d2} = {d1 + d2}");
        return d1 + d2;
    }

    // ── UI Helpers ───────────────────────────────────────────
    public void ShowMessage(string msg)
    {
        if (messageText != null) messageText.text = msg;
        Debug.Log("[MSG] " + msg);
    }

    public void UpdateHUD()
    {
        if (activePlayers == null || activePlayers.Count == 0) return;
        Player p = activePlayers[currentIndex];
        if (currentPlayerText != null) currentPlayerText.text = p.playerName + "'s turn";
        if (currentMoneyText  != null) currentMoneyText.text  = "$" + p.money;
    }

    // TODO (Lesson 5): StartTurn(), MovePlayer()
    // TODO (Lesson 6): ResolveSpace(), OfferPurchase(), ChargeRent()
    // TODO (Lesson 7): NextTurn(), EliminatePlayer(), GameOver()
}

Part 3 · Player Token Setup (15 min)

Setting Up Player Tokens

Each player needs a visible token on the board and a Player.cs script holding their data. Follow these steps for each player (2–3 players minimum).

1
Create a token GameObject

GameObject → 3D Object → Capsule. Rename it Player1. Set Scale to (0.5, 0.5, 0.5) so it sits neatly on the board squares without overlapping.

Scale: X 0.5 · Y 0.5 · Z 0.5  |  Position: same as Space_00_Go
2
Attach Player.cs

With Player1 selected in the Hierarchy, click Add Component in the Inspector → search for "Player" → click Player (Script). The component appears with all public fields exposed.

3
Fill in the Inspector fields

Set playerName to your player's name, money to 1500, playerColour to a distinctive colour. Drag the Capsule's MeshRenderer into the tokenRenderer slot — this lets the script tint the token automatically on Start.

playerName: "Nomad"  |  money: 1500  |  position: 0  |  isActive: ✓
4
Assign the token self-reference

Drag the Player1 GameObject from the Hierarchy into the token field in its own Inspector. This lets GameManager move the token by calling player.token.transform.position.

5
Duplicate for additional players

Select Player1 in the Hierarchy → Ctrl+D → rename to Player2 → change playerName and playerColour. Move slightly on the Y axis (e.g. Y = 0.05) so tokens don't clip through each other when on the same space.

Player2 position offset: Y + 0.05  |  Player3: Y + 0.10

Part 4 · Wiring GameManager (10 min)

Connecting Everything Together

The GameManager is the hub — it needs references to every board space and every player to function. This is done entirely through the Inspector (no code changes).

1
Create the GameManager GameObject

In the Hierarchy: right-click → Create Empty → rename to GameManager. Position is irrelevant (it's invisible) — place it at the origin (0,0,0).

2
Attach GameManager.cs

Add Component → GameManager. The Inspector now shows all the public fields: boardSpaces[], activePlayers, goSalary, and the UI text references.

3
Populate boardSpaces[ ]

Click the lock icon (🔒) on the Inspector to pin it. In the Hierarchy, select all space GameObjects in order (click Space_00, then Shift+click Space_23). Drag them all into the boardSpaces array field. Confirm that Element 0 = Space_00_Go.

boardSpaces: Size 24  |  Element 0 = Space_00_Go  |  Element 23 = last space
4
Add players to activePlayers

Set the activePlayers list size to 2 (or 3). Drag Player1 into Element 0, Player2 into Element 1. The first player in this list takes the first turn.

5
Press Play — check the Console

Hit the Play button. The Console should show no red errors — only the welcome message: "Welcome to Dice District! [Player1] goes first. Roll the dice!". If you see red errors about empty arrays, check Steps 3–4 above.

Expected Console output: [MSG] Welcome to Dice District! Nomad goes first. Roll the dice!
Debug tip: If you see "boardSpaces array is empty" in red, it means you forgot to drag the spaces in, or you dragged the parent empty GameObject instead of the individual space objects. The boardSpaces array should have exactly as many elements as you have spaces on your board.

Scene saved? Press Ctrl+S before ending the lesson. Unity does not auto-save. If Unity crashes with an unsaved scene you will lose all the Inspector wiring — the scripts remain but every drag-and-drop connection is gone.

Common Errors

Red error: "boardSpaces is empty"
Forgot to drag spaces into the array, or dragged the parent folder instead of each space.

"The type or namespace 'TMP_Text' could not be found"
TextMeshPro not imported. Window → TextMeshPro → Import TMP Essential Resources.

Tokens not visible in Game view
Camera position wrong, or tokens are inside the board tiles (Y = 0). Raise token Y to 0.15+.
