Part 1 · The Decision Engine (10 min)

What Happens When You Land?

Every space on your board does something different when a player lands on it. The ResolveSpace() method is the game's decision engine — it reads the landed space's type and triggers the correct action. There are at most five possible branches, and only one fires per landing.

Part 3 · Writing the Code (30 min)

Four Methods to Implement

Open GameManager.cs. You will add four methods and update one line in MovePlayer(). Copy and read each method in order — they call each other.

First: In MovePlayer(), find the line NextTurn(); // temporary and replace it with ResolveSpace(p, boardSpaces[newPosition]); — this connects movement to space resolution.
Method 1 — The decision engine GameManager.cs 
// Called by MovePlayer() after every move.
// Reads the space type and routes to the correct action.
public void ResolveSpace(Player p, BoardSpace space)
{
    switch (space.spaceType)
    {
        case SpaceType.Go:
            // Salary for landing directly on Go (passing already handled in MovePlayer)
            p.AddMoney(goSalary);
            ShowMessage($"{p.playerName} landed on Go! Collected ${goSalary}.");
            UpdateHUD();
            NextTurn();
            break;

        case SpaceType.Property:
            if (space.IsAvailable())
            {
                OfferPurchase(p, space);   // shows Buy/Skip buttons
            }
            else if (space.IsOwnedByOther(p))
            {
                ChargeRent(p, space);      // deducts rent, advances turn
            }
            else
            {
                // Player owns this space — no action needed
                ShowMessage($"{p.playerName} landed on their own property: {space.spaceName}.");
                NextTurn();
            }
            break;

        case SpaceType.Tax:
            bool canAfford = p.ChargeMoney(space.taxAmount);
            ShowMessage($"{p.playerName} paid ${space.taxAmount} tax on {space.spaceName}.");
            UpdateHUD();
            if (!canAfford)
                CheckBankruptcy(p);   // implemented in Lesson 7
            else
                NextTurn();
            break;

        case SpaceType.Chance:
            // Stub — full Chance card system added in Lesson 8
            ShowMessage($"{p.playerName} drew a Chance card... nothing happens yet!");
            NextTurn();
            break;
    }
}
Method 2 — Offer a property for purchase GameManager.cs 
// Shows the Buy and Skip buttons and stores the pending offer.
// The actual purchase happens when the player clicks Buy (see BuyProperty below).
private BoardSpace pendingPurchase = null; // which space is being offered

public void OfferPurchase(Player p, BoardSpace space)
{
    waitingForInput = true;   // blocks the Roll Dice button
    pendingPurchase = space;

    ShowMessage($"{space.spaceName} is available! "
              + $"Buy for ${space.purchasePrice}? "
              + $"You have ${p.money}.");

    // Show the Buy and Skip buttons in the UI
    if (buyButton  != null) buyButton.gameObject.SetActive(true);
    if (skipButton != null) skipButton.gameObject.SetActive(true);
}

// Called by the Buy button's OnClick event.
public void BuyProperty()
{
    if (pendingPurchase == null) return;

    Player p = activePlayers[currentIndex];

    if (p.money >= pendingPurchase.purchasePrice)
    {
        p.ChargeMoney(pendingPurchase.purchasePrice);
        p.AddProperty(pendingPurchase);   // sets space.owner = this player
        ShowMessage($"{p.playerName} bought {pendingPurchase.spaceName} "
                  + $"for ${pendingPurchase.purchasePrice}!");
    }
    else
    {
        ShowMessage($"{p.playerName} can't afford {pendingPurchase.spaceName}.");
    }

    HidePurchaseButtons();
    UpdateHUD();
    NextTurn();
}

// Called by the Skip button's OnClick event.
public void SkipPurchase()
{
    if (pendingPurchase == null) return;
    ShowMessage($"{activePlayers[currentIndex].playerName} chose not to buy {pendingPurchase.spaceName}.");
    HidePurchaseButtons();
    NextTurn();
}

private void HidePurchaseButtons()
{
    waitingForInput = false;
    pendingPurchase = null;
    if (buyButton  != null) buyButton.gameObject.SetActive(false);
    if (skipButton != null) skipButton.gameObject.SetActive(false);
}
Method 3 — Charge rent to another player GameManager.cs 
public void ChargeRent(Player payer, BoardSpace space)
{
    Player owner = space.owner;
    int    rent  = space.rentAmount;

    // Transfer money: payer → owner
    bool canAfford = payer.ChargeMoney(rent);
    owner.AddMoney(rent);

    ShowMessage($"{payer.playerName} paid ${rent} rent to "
              + $"{owner.playerName} for {space.spaceName}.");
    UpdateHUD();

    if (!canAfford)
        CheckBankruptcy(payer);  // implemented in Lesson 7
    else
        NextTurn();
}

// Bankruptcy stub — prevents compile errors until Lesson 7
private void CheckBankruptcy(Player p)
{
    ShowMessage($"{p.playerName} may be bankrupt! (Full check in Lesson 7)");
    Debug.Log($"Bankruptcy check: {p.playerName} has ${p.money}");
    NextTurn();  // temporary — replaced in Lesson 7
}

Common Errors

Buy buttons stay visible permanently
Forgot to call HidePurchaseButtons() — check that both BuyProperty() and SkipPurchase() call it before NextTurn().

Roll Dice still works mid-purchase
waitingForInput not set — confirm OfferPurchase() sets it to true and HidePurchaseButtons() sets it back to false.

"Object reference not set" on owner
space.owner was never initialised — the BoardSpace Inspector's Owner field must be left as None (null), not accidentally dragged to a player.
