// Add these two fields at the top of GameManager:
private int lastD1, lastD2;

// Replace the existing RollDice() body with:
public int RollDice()
{
    lastD1 = Random.Range(1, 7);
    lastD2 = Random.Range(1, 7);
    int total = lastD1 + lastD2;

    if (die1Text != null)
        die1Text.text = DiceFace(lastD1);
    if (die2Text != null)
        die2Text.text = DiceFace(lastD2);

    Debug.Log($"Rolled: {lastD1} + {lastD2} = {total}");
    return total;
}

// Helper — maps a value to a Unicode dice face
private string DiceFace(int value)
{
    string[] faces = { "⚀", "⚁", "⚂", "⚃", "⚄", "⚅" };
    return (value >= 1 && value <= 6)
        ? faces[value - 1] : value.ToString();
}

🃏 Chance Card System
~20 min ▶

Replaces the Chance stub from Lesson 6 with a real shuffled deck of 8 event cards. When a player lands on a Chance space, one card is drawn — it might give money, take money, or move the player. The deck reshuffles when exhausted.

  1. Define your 8 chance events in the Inspector as a struct array (or use the code below with a ChanceCard inner class).
  2. Shuffle the deck using Fisher-Yates on Start. Draw from the front; reshuffle when empty.
  3. Apply the card's effect (money change or position jump) then call NextTurn().
Chance card system — add to GameManager.cs 
// Inner class — add INSIDE GameManager class body
[System.Serializable]
public class ChanceCard
{
    public string description;
    public int    moneyChange; // positive = gain, negative = lose
    public int    moveToSpace; // -1 = no move
}

// Add to GameManager fields:
[Header("Chance Cards")]
public ChanceCard[] chanceCards;     // fill in Inspector
private List<ChanceCard> chanceDeck = new List<ChanceCard>();

// Add to Start():
ShuffleChanceDeck();

// New methods:
private void ShuffleChanceDeck()
{
    chanceDeck = new List<ChanceCard>(chanceCards);
    for (int i = chanceDeck.Count - 1; i > 0; i--)
    {
        int j = Random.Range(0, i + 1);
        (ChanceCard tmp, chanceDeck[i], chanceDeck[j]) =
            (chanceDeck[i], chanceDeck[j], chanceDeck[i]);
        chanceDeck[i] = chanceDeck[j];
        chanceDeck[j] = tmp;
    }
}

private void DrawChanceCard(Player p)
{
    if (chanceDeck.Count == 0) ShuffleChanceDeck();
    ChanceCard card = chanceDeck[0];
    chanceDeck.RemoveAt(0);

    ShowMessage($"Chance: {card.description}");

    if (card.moneyChange != 0)
    {
        if (card.moneyChange > 0) p.AddMoney(card.moneyChange);
        else { bool ok = p.ChargeMoney(-card.moneyChange); if(!ok) { CheckBankruptcy(p); return; } }
    }
    if (card.moveToSpace >= 0)
        p.token.transform.position =
            boardSpaces[card.moveToSpace].transform.position + Vector3.up * 0.6f;

    UpdateHUD();
    NextTurn();
}

// In ResolveSpace() case SpaceType.Chance — replace stub with:
// DrawChanceCard(p);

In the Inspector, set chanceCards array size to 8 and fill in each card's description, moneyChange, and moveToSpace (use −1 for no move).

🏃 Animated Token Movement
~20 min ▶

Replace the instant position snap in MovePlayer() with a smooth Lerp animation. The token glides to its target over 0.4 seconds. A Coroutine handles the timing so ResolveSpace() only fires after the animation completes.

  1. Remove the instant p.token.transform.position = targetPos; line from MovePlayer().
  2. Replace it with StartCoroutine(MoveTokenSmoothly(p, targetPos)); and return early from MovePlayer() — the Coroutine will call ResolveSpace() when done.
  3. Add the Coroutine method below to GameManager.
Smooth token movement Coroutine — GameManager.cs 
// Add moveSpeed field at top of GameManager:
public float moveSpeed = 0.4f; // seconds per move

// Replace the token movement section inside MovePlayer() with:
Vector3 targetPos = boardSpaces[newPosition].transform.position;
targetPos.y += 0.6f;

if (p.token != null)
    StartCoroutine(MoveTokenSmoothly(p, targetPos, boardSpaces[newPosition]));
else
    ResolveSpace(p, boardSpaces[newPosition]);
return; // ResolveSpace will be called by the Coroutine

// Add this Coroutine to GameManager:
private IEnumerator MoveTokenSmoothly(
    Player p, Vector3 target, BoardSpace landingSpace)
{
    Vector3 start   = p.token.transform.position;
    float   elapsed = 0f;

    while (elapsed < moveSpeed)
    {
        elapsed += Time.deltaTime;
        float t = Mathf.Clamp01(elapsed / moveSpeed);
        // Ease-in-out curve
        t = t * t * (3f - 2f * t);
        p.token.transform.position = Vector3.Lerp(start, target, t);
        yield return null;
    }

    p.token.transform.position = target;
    ResolveSpace(p, landingSpace);
}
⚠ Important: You must add using System.Collections; at the top of GameManager.cs for IEnumerator to compile.

🎨 Property Group Bonus
~20 min ▶

When a player owns all properties of the same colour group, rent is doubled for that group. This is the core strategic mechanic of Monopoly. It requires iterating boardSpaces to count group members and checking against the player's owned list.

  1. In ChargeRent(), before deducting rent, call a new helper OwnsFullGroup().
  2. OwnsFullGroup() counts all Property spaces with the same groupColour as the landed space, then checks if the rent-receiver owns all of them.
  3. If yes, multiply rent by 2 and show an appropriate message.
Group bonus — add to GameManager.cs 
private bool OwnsFullGroup(Player player, BoardSpace space)
{
    int   groupTotal  = 0;
    int   playerOwned = 0;
    Color group       = space.groupColour;

    foreach (BoardSpace s in boardSpaces)
    {
        if (s.spaceType == SpaceType.Property && s.groupColour == group)
        {
            groupTotal++;
            if (s.owner == player) playerOwned++;
        }
    }

    return groupTotal > 0 && playerOwned == groupTotal;
}

// In ChargeRent(), replace:
//   int rent = space.rentAmount;
// with:
int rent = OwnsFullGroup(owner, space)
    ? space.rentAmount * 2
    : space.rentAmount;

if (OwnsFullGroup(owner, space))
    ShowMessage($"{owner.playerName} owns the full group — double rent! ${rent}");

AI turn Coroutine — GameManager.cs 
// Add to end of NextTurn(), after ShowMessage():
if (activePlayers[currentIndex].isAI)
    StartCoroutine(AITurn());

// Add this Coroutine:
private IEnumerator AITurn()
{
    yield return new WaitForSeconds(1.2f); // pause for legibility
    StartTurn();

    // If the AI landed on a purchasable property,
    // auto-decide: buy if money > 2 × price (simple heuristic)
    yield return new WaitForSeconds(0.4f);
    if (waitingForInput && pendingPurchase != null)
    {
        Player ai = activePlayers[currentIndex];
        if (ai.money >= pendingPurchase.purchasePrice * 2)
        {
            ShowMessage($"AI buys {pendingPurchase.spaceName}!");
            BuyProperty();
        }
        else
        {
            ShowMessage($"AI skips {pendingPurchase.spaceName}.");
            SkipPurchase();
        }
    }
}
👣 Step-by-Step Token Movement
~30 min ▶

Instead of jumping directly to the final space, the token visits every intermediate space one step at a time with a short pause. This makes the movement count visible and satisfying — especially when passing Go.

  1. Replace MoveTokenSmoothly() with MoveTokenStepByStep() — a Coroutine that loops through each intermediate index, snapping or lerping to each one with a 0.18s pause.
  2. Detect Go crossing during the step loop, not at the end — award salary the moment the token steps onto space 0.
  3. Call ResolveSpace() only after the final step lands.
Step-by-step movement Coroutine — GameManager.cs 
private IEnumerator MoveTokenStepByStep(
    Player p, int steps, int startPos)
{
    int pos = startPos;
    for (int i = 0; i < steps; i++)
    {
        pos = (pos + 1) % boardSpaces.Length;

        // Detect passing Go mid-step
        if (pos == 0 && i < steps - 1)
        {
            p.AddMoney(goSalary);
            ShowMessage($"{p.playerName} passed Go! +${goSalary}");
            UpdateHUD();
        }

        // Snap token to this step
        Vector3 stepTarget = boardSpaces[pos].transform.position;
        stepTarget.y += 0.6f;
        p.token.transform.position = stepTarget;

        yield return new WaitForSeconds(0.18f);
    }

    // All steps done — land and resolve
    p.position = pos;
    UpdateHUD();
    Debug.Log($"{p.playerName} stepped to {boardSpaces[pos].spaceName}");
    ResolveSpace(p, boardSpaces[pos]);
}
